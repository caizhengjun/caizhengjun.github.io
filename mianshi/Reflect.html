      # 1.1 Animal 类

      public class Animal {
  public String name = "Dog";
  private int age = 30;

  //默认无参构造函数
  public Animal() {
    System.out.println("Animal");
  }

  //带参数的构造函数
  public Animal(String name, int age) {
    System.out.println(name + "," + age);
  }

  //公开 方法  返回类型和参数均有
  public String sayName(String name) {
    return "Hello," + name;
  }
}

      # 2. 反射的4种创建方式

      public class ReflectTest {
  public static void main(String[] args) throws Exception {

    // 反射获取对象的4种方式(我的理解)
    // 1 靠爷爷
    Class<?> c1 = Class.forName("com.codereview.demo.reflect.Animal");
    // 2 靠爸爸
    Class<?> c2 = ReflectTest.class.getClassLoader().loadClass("com.codereview.demo.reflect.Animal");
    // 3 靠天赋
    Class<Animal> c3 = Animal.class;
    // 4 靠努力
    Animal animal = new Animal();
    Class<? extends Animal> c4 = animal.getClass();
  }
}

//c1 = class com.codereview.demo.reflect.Animal
//c2 = class com.codereview.demo.reflect.Animal
//c3 = class com.codereview.demo.reflect.Animal
//c4 = class com.codereview.demo.reflect.Animal
